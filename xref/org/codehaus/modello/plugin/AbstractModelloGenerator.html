59  public abstract class AbstractModelloGenerator
60      extends AbstractLogEnabled
61      implements ModelloGenerator, Contextualizable

64  
65      private File outputDirectory;

68  
69      private boolean packageWithVersion;
70  
71      private String encoding;
72  
73      private BuildContext buildContext;

75      protected void initialize( Model model, Properties parameters )
76          throws ModelloException
77      {
78          this.model = model;
79  
80          outputDirectory = new File( getParameter( parameters, ModelloParameterConstants.OUTPUT_DIRECTORY ) );
81  
82          String version = getParameter( parameters, ModelloParameterConstants.VERSION );
83  
84          generatedVersion = new Version( version );
85  
86          packageWithVersion =
87              Boolean.valueOf( getParameter( parameters, ModelloParameterConstants.PACKAGE_WITH_VERSION ) );
88  
89          encoding = parameters.getProperty( ModelloParameterConstants.ENCODING );
90      }

117     protected String getHeader()
118     {
119         String version = getClass().getPackage().getImplementationVersion();
120         return "=================== DO NOT EDIT THIS FILE ====================\n"
121             + "Generated by Modello" + ( ( version == null ) ? "" : ( ' ' + version ) ) + ",\n"
122             + "any modifications will be overwritten.\n"
123             + "==============================================================";
124     }
125 
126     protected boolean isClassInModel( String fieldType, Model model )
127     {
128         try
129         {
130             return model.getClass( fieldType, generatedVersion ) != null;
131         }
132         catch ( Exception e )
133         {
134         }
135 
136         return false;
137     }
138 
139     /**
140      * Return the child fields of this class.
141      * @param modelClass current class
142      * @return the list of fields of this class
143      */
144     protected List<ModelField> getFieldsForClass( ModelClass modelClass )
145     {
146         List<ModelField> fields = new ArrayList<ModelField>();
147 
148         while ( modelClass != null )
149         {
150             fields.addAll( modelClass.getFields( getGeneratedVersion() ) );
151 
152             String superClass = modelClass.getSuperClass();
153             if ( superClass != null )
154             {
155                 modelClass = getModel().getClass( superClass, getGeneratedVersion() );
156             }
157             else
158             {
159                 modelClass = null;
160             }
161         }
162 
163         return fields;
164     }
165 
166     protected boolean isInnerAssociation( ModelField field )
167     {
168         return field instanceof ModelAssociation
169             && isClassInModel( ( (ModelAssociation) field ).getTo(), getModel() );
170     }

172     protected boolean isMap( String fieldType )
173     {
174         return ModelDefault.MAP.equals( fieldType ) || ModelDefault.PROPERTIES.equals( fieldType );
175     }
176 
177     protected boolean isCollection( String fieldType )
178     {
179         return ModelDefault.LIST.equals( fieldType ) || ModelDefault.SET.equals( fieldType );
180     }

182     protected String capitalise( String str )
183     {
184         if ( StringUtils.isEmpty( str ) )
185         {
186             return str;
187         }
188 
189         return new StringBuilder( str.length() )
190             .append( Character.toTitleCase( str.charAt( 0 ) ) )
191             .append( str.substring( 1 ) )
192             .toString();
193     }
194 
195     public static String singular( String name )
196     {
197         if ( StringUtils.isEmpty( name ) )
198         {
199             return name;
200         }
201 
202         if ( name.endsWith( "ies" ) )
203         {
204             return name.substring( 0, name.length() - 3 ) + "y";
205         }
206         else if ( name.endsWith( "es" ) && name.endsWith( "ches" ) )
207         {
208             return name.substring( 0, name.length() - 2 );
209         }
210         else if ( name.endsWith( "xes" ) )
211         {
212             return name.substring( 0, name.length() - 2 );
213         }
214         else if ( name.endsWith( "s" ) && ( name.length() != 1 ) )
215         {
216             return name.substring( 0, name.length() - 1 );
217         }
218 
219         return name;
220     }
221 
222     public static String uncapitalise( String str )
223     {
224         if ( StringUtils.isEmpty( str ) )
225         {
226             return str;
227         }
228 
229         return new StringBuilder( str.length() )
230             .append( Character.toLowerCase( str.charAt( 0 ) ) )
231             .append( str.substring( 1 ) )
232             .toString();
233     }

277     public void contextualize( Context ctx )
278         throws ContextException
279     {
280         PlexusContainer plexus = (PlexusContainer) ctx.get( PlexusConstants.PLEXUS_KEY );
281 
282         try
283         {
284             buildContext = (BuildContext) plexus.lookup( BuildContext.class.getName() );
285         }
286         catch( ComponentLookupException e )
287         {
288             throw new ContextException( "Unable to lookup required component", e );
289         }
290     }
291 
292     protected BuildContext getBuildContext()
293     {
294         return buildContext;
295     }

297     protected Writer newWriter( Path path ) throws IOException
298     {
299         Charset charset = getEncoding() != null ? Charset.forName( getEncoding() ) : Charset.defaultCharset();
300         return newWriter( path, charset );
301     }
302 
303     protected Writer newWriter( Path path, Charset charset ) throws IOException
304     {
305         CachingWriter cachingWriter = new CachingWriter( path, charset );
306         return new FilterWriter( cachingWriter )
307         {
308             @Override
309             public void close() throws IOException
310             {
311                 super.close();
312                 if ( cachingWriter.isModified() )
313                 {
314                     getBuildContext().refresh( path.toFile() );
315                 }
316                 else
317                 {
318                     getLogger().debug( "The contents of the file " + path + " matches, skipping writing file." );
319                 }
320             }
321         };
322     }
