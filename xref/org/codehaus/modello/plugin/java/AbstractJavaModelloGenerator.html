62  /**
63   * AbstractJavaModelloGenerator - similar in scope to {@link AbstractModelloGenerator} but with features that
64   * java generators can use.
65   *
66   * @author <a href="mailto:joakim@erdfelt.com">Joakim Erdfelt</a>
67   */
68  public abstract class AbstractJavaModelloGenerator
69      extends AbstractModelloGenerator
70  {
71      private Optional<Integer> javaSource;
72  
73      protected boolean domAsXpp3 = true;
74  
75      protected void initialize( Model model, Properties parameters )
76          throws ModelloException
77      {
78          super.initialize( model, parameters );
79  
80          javaSource = Optional.ofNullable( getParameter( parameters, ModelloParameterConstants.OUTPUT_JAVA_SOURCE, null ) ).map( Integer::valueOf );
81  
82          domAsXpp3 = !"false".equals( parameters.getProperty( ModelloParameterConstants.DOM_AS_XPP3 ) );
83      }

85      /**
86       * Create a new java source file writer, with configured encoding.
87       *
88       * @param packageName the package of the source file to create
89       * @param className the class of the source file to create
90       * @return a JSourceWriter with configured encoding
91       * @throws IOException
92       */
93      protected JSourceWriter newJSourceWriter( String packageName, String className )
94          throws IOException
95      {
96          String directory = packageName.replace( '.', File.separatorChar );
97  
98          File f = new File( new File( getOutputDirectory(), directory ), className + ".java" );
99  
100         if ( !f.getParentFile().exists() )
101         {
102             f.getParentFile().mkdirs();
103         }
104 
105         return new JSourceWriter( newWriter( f.toPath() ) );
106     }

130     protected void suppressAllWarnings( Model objectModel, JStructure structure )
131     {
132         JavaModelMetadata javaModelMetadata = (JavaModelMetadata) objectModel.getMetadata( JavaModelMetadata.ID );
133 
134         if ( hasJavaSourceSupport( 5 ) && javaModelMetadata.isSuppressAllWarnings() )
135         {
136             structure.appendAnnotation( "@SuppressWarnings( \"all\" )" );
137         }

140     protected void addModelImports( JClass jClass, BaseElement baseElem )
141         throws ModelloException
142     {
143         String basePackageName = null;
144         if ( baseElem instanceof ModelType )
145         {
146             basePackageName = ( (ModelType) baseElem ).getPackageName( isPackageWithVersion(), getGeneratedVersion() );
147         }
148 
149         // import interfaces
150         for ( ModelInterface modelInterface : getModel().getInterfaces( getGeneratedVersion() ) )
151         {
152             addModelImport( jClass, modelInterface, basePackageName );
153         }
154 
155         // import classes
156         for ( ModelClass modelClass : getClasses( getModel() ) )
157         {
158             addModelImport( jClass, modelClass, basePackageName );
159         }
160     }
161 
162     protected void addModelImport( JClass jClass, ModelType modelType, String basePackageName )
163     {
164         String packageName = modelType.getPackageName( isPackageWithVersion(), getGeneratedVersion() );
165 
166         if ( !packageName.equals( basePackageName ) )
167         {
168             jClass.addImport( packageName + '.' + modelType.getName() );
169         }
170     }

177     protected String getDefaultValue( ModelAssociation association )
178     {
179         String value = association.getDefaultValue();
180 
181         if ( hasJavaSourceSupport( 5 ) )
182         {
183             value = StringUtils.replaceOnce( StringUtils.replaceOnce( value, "/*", "" ), "*/", "" );
184         }
185 
186         return value;

189     protected static final String DEFAULT_DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSS";
190 
191     protected String getJavaDefaultValue( ModelField modelField )
192         throws ModelloException
193     {
194         String type = modelField.getType();
195         String value = modelField.getDefaultValue();
196 
197         if ( "String".equals( type ) )
198         {
199             return '"' + escapeStringLiteral( value ) + '"';
200         }
201         else if ( "char".equals( type ) )
202         {
203             return '\'' + escapeStringLiteral( value ) + '\'';
204         }
205         else if ( "long".equals( type ) )
206         {
207             return value + 'L';
208         }
209         else if ( "float".equals( type ) )
210         {
211             return value + 'f';
212         }
213         else if ( "Date".equals( type ) )
214         {
215             DateFormat format = new SimpleDateFormat( DEFAULT_DATE_FORMAT, Locale.US );
216             try
217             {
218                 Date date = format.parse( value );
219                 return "new java.util.Date( " + date.getTime() + "L )";
220             }
221             catch ( ParseException pe )
222             {
223                 throw new ModelloException( "Unparseable default date: " + value, pe );
224             }
225         }
226         else if ( value != null && value.length() > 0 )
227         {
228             boolean useJava5 = hasJavaSourceSupport( 5 );
229             if ( "Character".equals( type ) && !value.contains( type ) )
230             {
231                 return newPrimitiveWrapper( type, "'" + escapeStringLiteral( value ) + "'", useJava5 );
232             }
233             else if ( "Boolean".equals( type ) && !value.contains( type ) )
234             {
235                 return newPrimitiveWrapper( type, value, true );
236             }
237             else if ( "Byte".equals( type ) && !value.contains( type ) )
238             {
239                 return newPrimitiveWrapper( type, "(byte) " + value, useJava5 );
240             }
241             else if ( "Short".equals( type ) && !value.contains( type ) )
242             {
243                 return newPrimitiveWrapper( type, "(short) " + value, useJava5 );
244             }
245             else if ( "Integer".equals( type ) && !value.contains( type ) )
246             {
247                 return newPrimitiveWrapper( type, value, useJava5 );
248             }
249             else if ( "Long".equals( type ) && !value.contains( type ) )
250             {
251                 return newPrimitiveWrapper( type, value + 'L', useJava5 );
252             }
253             else if ( "Float".equals( type ) && !value.contains( type ) )
254             {
255                 return newPrimitiveWrapper( type, value + 'f', useJava5 );
256             }
257             else if ( "Double".equals( type ) && !value.contains( type ) )
258             {
259                 return newPrimitiveWrapper( type, value, useJava5 );
260             }
261         }
262 
263         return value;
264     }
265 
266     private String newPrimitiveWrapper( String type, String value, boolean useJava5 )
267     {
268         if ( useJava5 )
269         {
270             return type + ".valueOf( " + value + " )";
271         }
272         else
273         {
274             return "new " + type + "( " + value + " )";
275         }
276     }

278     private String escapeStringLiteral( String str )
279     {
280         StringBuilder buffer = new StringBuilder( str.length() + 32 );
281 
282         for ( int i = 0, n = str.length(); i < n; i++ )
283         {
284             char c = str.charAt( i );
285             switch ( c )
286             {
287                 case '\0':
288                     buffer.append( "\\0" );
289                     break;
290                 case '\t':
291                     buffer.append( "\\t" );
292                     break;
293                 case '\r':
294                     buffer.append( "\\r" );
295                     break;
296                 case '\n':
297                     buffer.append( "\\n" );
298                     break;
299                 case '\\':
300                     buffer.append( "\\\\" );
301                     break;
302                 default:
303                     buffer.append( c );
304             }
305         }
306 
307         return buffer.toString();
308     }

310     protected String getValueChecker( String type, String value, ModelField field )
311         throws ModelloException
312     {
313         String retVal;
314         if ( "boolean".equals( type ) || "double".equals( type ) || "float".equals( type ) || "int".equals( type )
315             || "long".equals( type ) || "short".equals( type ) || "byte".equals( type ) || "char".equals( type ) )
316         {
317             retVal = "if ( " + value + " != " + getJavaDefaultValue( field ) + " )";
318         }
319         else if ( ModelDefault.LIST.equals( type ) || ModelDefault.SET.equals( type )
320             || ModelDefault.MAP.equals( type ) || ModelDefault.PROPERTIES.equals( type ) )
321         {
322             retVal = "if ( ( " + value + " != null ) && ( " + value + ".size() > 0 ) )";
323         }
324         else if ( "String".equals( type ) && field.getDefaultValue() != null )
325         {
326             retVal = "if ( ( " + value + " != null ) && !" + value + ".equals( \"" + field.getDefaultValue() + "\" ) )";
327         }
328         else if ( "Date".equals( type ) && field.getDefaultValue() != null )
329         {
330             retVal = "if ( ( " + value + " != null ) && !" + value + ".equals( " + getJavaDefaultValue( field ) + " ) )";
331         }
332         else
333         {
334             retVal = "if ( " + value + " != null )";
335         }
336         return retVal;
337     }

339     protected List<ModelClass> getClasses( Model model )
340     {
341         List<ModelClass> modelClasses = new ArrayList<ModelClass>();
342 
343         for ( ModelClass modelClass : model.getClasses( getGeneratedVersion() ) )
344         {
345             if ( isRelevant( modelClass ) )
346             {
347                 modelClasses.add( modelClass );
348             }
349         }
350 
351         return modelClasses;
352     }
353 
354     protected boolean isRelevant( ModelClass modelClass )
355     {
356         return isJavaEnabled( modelClass ) && !isTrackingSupport( modelClass );
357     }
358 
359     protected boolean isJavaEnabled( ModelClass modelClass )
360     {
361         JavaClassMetadata javaClassMetadata = (JavaClassMetadata) modelClass.getMetadata( JavaClassMetadata.ID );
362         return javaClassMetadata.isEnabled();
363     }
364 
365     protected boolean isTrackingSupport( ModelClass modelClass )
366     {
367         ModelClassMetadata modelClassMetadata = (ModelClassMetadata) modelClass.getMetadata( ModelClassMetadata.ID );
368         if ( StringUtils.isNotEmpty( modelClassMetadata.getLocationTracker() ) )
369         {
370             return true;
371         }
372         if ( StringUtils.isNotEmpty( modelClassMetadata.getSourceTracker() ) )
373         {
374             return true;
375         }
376         return false;
377     }
