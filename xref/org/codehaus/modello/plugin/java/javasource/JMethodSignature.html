78  /**
79   * A class which holds information about the signature
80   * of a JMethod.
81   *
82   * The code in this package was modelled after the Java Reflection API
83   * as much as possible to reduce the learning curve.
84   *
85   * @author <a href="mailto:kvisco@intalio.com">Keith Visco</a>
86   **/
87  public final class JMethodSignature
88  {
89  
90      /**
91       * The set of modifiers for this JMethod
92       **/
93      private JModifiers modifiers = null;
94  
95      /**
96       * The return type of this Method
97       **/
98      private JType returnType = null;
99  
100     /**
101      * The name of this method
102      **/
103     private String name = null;
104 
105 
106     /**
107      * The list of parameters of this JMethodSignature in declared
108      * order
109      **/
110     private Map<String, JParameter> params = null;
111 
112     /**
113      * The JavaDoc comment for this method signature.
114      **/
115     private JDocComment jdc = null;
116 
117     /**
118      * The exceptions that this method throws
119      **/
120     private List<JClass> exceptions = null;
121 
122     /**
123      * Creates a new method with the given name and return type.
124      * For "void" return types, simply pass in null as the returnType
125      *
126      * @param name, the method name. Must not be null.
127      * @param returnType the return type of the method. May be null.
128      **/
129     public JMethodSignature( String name, JType returnType )
130     {
131 
132         if ( ( name == null ) || ( name.length() == 0 ) )
133         {
134             String err = "The method name must not be null or zero-length";
135             throw new IllegalArgumentException( err );
136         }
137 
138         this.jdc = new JDocComment();
139         this.returnType = returnType;
140         this.name = name;
141         this.modifiers = new JModifiers();
142         this.params = new LinkedHashMap<>( 3 );
143         this.exceptions = new ArrayList<JClass>( 1 );
144     } //-- JMethodSignature
145 
146     /**
147      * Adds the given Exception to this JMethodSignature's throws clause.
148      *
149      * @param exp the JClass representing the Exception
150      **/
151     public void addException( JClass exp )
152     {
153 
154         if ( exp == null ) return;
155 
156         //-- make sure exception is not already added
157         String expClassName = exp.getName();
158         for ( JClass jClass  : exceptions )
159         {
160             if ( expClassName.equals( jClass.getName() ) ) return;
161         }
162         //-- add exception
163         exceptions.add( exp );
164 
165         //-- create comment
166         jdc.addDescriptor( JDocDescriptor.createExceptionDesc( expClassName, null ) );
167     } //-- addException
168 
169     /**
170      * Adds the given parameter to this JMethodSignature's list of
171      * parameters.
172      *
173      * @param parameter the parameter to add to the this Methods
174      * list of parameters.
175      * @throws java.lang.IllegalArgumentException when a parameter already
176      * exists for this Method with the same name as the new
177      * parameter.
178      **/
179     public void addParameter( JParameter parameter )
180         throws IllegalArgumentException
181     {
182 
183         if ( parameter == null ) return;
184 
185         String pName = parameter.getName();
186         //-- check current params
187         if ( params.get( pName ) != null )
188         {
189             StringBuilder err = new StringBuilder();
190             err.append( "A parameter already exists for this method, " );
191             err.append( name );
192             err.append( ", with the name: " );
193             err.append( pName );
194             throw new IllegalArgumentException( err.toString() );
195         }
196 
197 
198         params.put( pName, parameter );
199 
200         //-- create comment
201         jdc.addDescriptor( JDocDescriptor.createParamDesc( pName, null ) );
202 
203     } //-- addParameter
204 
205     /**
206      * Returns the exceptions that this JMethodSignature lists
207      * in it's throws clause.
208      *
209      * @return the exceptions that this JMethodSignature lists
210      * in it's throws clause.
211      **/
212     public JClass[] getExceptions()
213     {
214         return exceptions.toArray( new JClass[0] );
215     } //-- getExceptions
216 
217     /**
218      * Returns the JDocComment describing this JMethodSignature
219      *
220      * @return the JDocComment describing this JMethodSignature
221      **/
222     public JDocComment getJDocComment()
223     {
224         return this.jdc;
225     } //-- getJDocComment
226 
227     /**
228      * Returns the modifiers for this JMethodSignature.
229      *
230      * @return the modifiers for this JMethodSignature.
231      **/
232     public JModifiers getModifiers()
233     {
234         return this.modifiers;
235     } //-- getModifiers

237     /**
238      * Returns the name of the method.
239      *
240      * @return the name of the method.
241      **/
242     public String getName()
243     {
244         return this.name;
245     } //-- getName
246 
247     /**
248      * Returns the JParameter at the given index.
249      *
250      * @param index the index of the JParameter to return.
251      * @return the JParameter at the given index.
252      **/
253     public JParameter getParameter( int index )
254     {
255         Iterator<Map.Entry<String, JParameter>> paramIter = params.entrySet().iterator();
256         Map.Entry<String, JParameter> selected = null;
257         for( int i = 0; i <= index; i++)
258         {
259             selected = paramIter.next();
260         }
261         return selected.getValue();
262     } //-- getParameter
263 
264     /**
265      * Returns the set of JParameters for this JMethodSignature
266      * <BR>
267      * <B>Note:</B> the array is a copy, the params in the array
268      * are the actual references.
269      * @return the set of JParameters for this JMethod
270      **/
271     public synchronized JParameter[] getParameters()
272     {
273         return params.values().toArray( new JParameter[0] );
274     } //-- getParameters

277      * Returns the JType that represents the return type for the

279      *
280      * @return the JType that represents the return type for the
281      * method signature.
282      **/
283     public JType getReturnType()
284     {
285         return returnType;
286     } //-- getReturnType

302      * Sets the JModifiers for this method signature.
303      *
304      * @param modifiers the JModifiers for this method signature.

312     /**
313      * Prints the method signature. A semi-colon (end-of-statement
314      * terminator ';') will Not be printed.
315      *
316      * @param jsw the JSourceWriter to print to.
317      **/
318     public void print( JSourceWriter jsw )
319     {
320         print( jsw, true );
321     } //-- print
322 
323     /**
324      * Prints the method signature. A semi-colon (end-of-statement
325      * terminator ';') will Not be printed.
326      *
327      * @param jsw the JSourceWriter to print to.
328      * @param printJavaDoc a boolean that when true prints the JDocComment
329      * associated with this method signature.
330      **/
331     public void print( JSourceWriter jsw, boolean printJavaDoc )
332     {
333 
334         //------------/
335         //- Java Doc -/
336         //------------/
337 
338         if ( printJavaDoc ) jdc.print( jsw );
339 
340         //-----------------/
341         //- Method Source -/
342         //-----------------/
343 
344         jsw.write( modifiers.toString() );
345         if ( modifiers.toString().length() > 0 )
346         {
347             jsw.write( ' ' );
348         }
349         if ( returnType != null )
350         {
351             jsw.write( returnType );
352         }
353         else
354             jsw.write( "void" );
355         jsw.write( ' ' );
356         jsw.write( name );
357         jsw.write( '(' );
358 
359         //-- print parameters
360         if ( params.size() > 0 )
361         {
362             jsw.write( ' ' );
363 
364             Enumeration<JParameter> paramEnum = Collections.enumeration( params.values() );
365             jsw.write( paramEnum.nextElement() );
366             while( paramEnum.hasMoreElements() )
367             {
368                 jsw.write( ", " );
369                 jsw.write( paramEnum.nextElement() );
370             }
371             
372             jsw.write( ' ' );
373         }
374 
375         jsw.write( ')' );
376 
377         if ( exceptions.size() > 0 )
378         {
379             jsw.writeln();
380             jsw.write( "    throws " );
381             for ( int i = 0; i < exceptions.size(); i++ )
382             {
383                 if ( i > 0 ) jsw.write( ", " );
384                 JClass jClass = exceptions.get( i );
385                 jsw.write( jClass.getName() );
386             }
387         }
388     } //-- print

391     /**
392      * Returns the String representation of this JMethod,
393      * which is the method prototype.
394      * @return the String representation of this JMethod, which
395      * is simply the method prototype
396      **/
397     public String toString()
398     {
399 
400         StringBuilder sb = new StringBuilder();
401         if ( returnType != null )
402         {
403             sb.append( returnType );
404         }
405         else
406             sb.append( "void" );
407         sb.append( ' ' );
408         sb.append( name );
409         sb.append( '(' );
410 
411         //-- print parameters
412         if ( params.size() > 0 )
413         {
414             sb.append( ' ' );
415             
416             Enumeration<JParameter> paramEnum = Collections.enumeration( params.values() );
417             sb.append( paramEnum.nextElement().getType().getName() );
418             while( paramEnum.hasMoreElements() )
419             {
420                 sb.append( ", " );
421                 sb.append( paramEnum.nextElement().getType().getName() );
422             }
423             
424             sb.append( ' ' );
425         }
426 
427         sb.append( ')' );
428 
429         return sb.toString();
430     } //-- toString
431 
432     protected String[] getParameterClassNames()
433     {
434         List<String> names = new ArrayList<String>( params.size() );
435 
436         for ( JParameter param : params.values() )
437         {
438             JType jType = param.getType();
439             while ( jType.isArray() ) jType = jType.getComponentType();
440             if ( !jType.isPrimitive() )
441             {
442                 JClass jclass = (JClass) jType;
443                 names.add( jclass.getName() );
444             }
445         }
446 
447         return names.toArray( new String[0] );
448     } //-- getParameterClassNames
