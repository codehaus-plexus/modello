53  public class Xpp3WriterGenerator
54      extends AbstractXpp3Generator
55  {
56      private String extendedClassnameSuffix;
57  
58      protected void prepareLocationTracking( JClass jClass )
59      {
60          // NO OP
61      }
62  
63      protected void writeLocationTracking( JSourceCode sc, String name, String key )
64      {
65          if ( isLocationTracking() )
66          {
67              sc.add( "writeLocationTracking( " + name + ", " + key + ", serializer );" );
68          }
69      }
70  
71      @Override
72      protected void initialize( Model model, Properties parameters )
73          throws ModelloException
74      {
75          super.initialize( model, parameters );
76  
77          extendedClassnameSuffix = "Ex";
78          if ( isLocationTracking() )
79          {
80              extendedClassnameSuffix = parameters.getProperty( ModelloParameterConstants.EXTENDED_CLASSNAME_SUFFIX );
81          }
82      }
83  
84      public void generate( Model model, Properties parameters )
85          throws ModelloException
86      {
87          initialize( model, parameters );
88  
89          try
90          {
91              generateXpp3Writer();
92          }
93          catch ( IOException ex )
94          {
95              throw new ModelloException( "Exception while generating XPP3 Writer.", ex );
96          }
97      }
98  
99      private void generateXpp3Writer()
100         throws ModelloException, IOException
101     {
102         Model objectModel = getModel();
103 
104         String packageName = objectModel.getDefaultPackageName( isPackageWithVersion(), getGeneratedVersion() )
105             + ".io.xpp3";
106 
107         String marshallerName = getFileName( "Xpp3Writer" + ( isLocationTracking() ? extendedClassnameSuffix : "" ) );
108 
109         JSourceWriter sourceWriter = newJSourceWriter( packageName, marshallerName );
110 
111         JClass jClass = new JClass( packageName + '.' + marshallerName );
112         initHeader( jClass );
113         suppressAllWarnings( objectModel, jClass );
114 
115         jClass.addImport( "org.codehaus.plexus.util.xml.pull.XmlSerializer" );
116         jClass.addImport( "org.codehaus.plexus.util.xml.pull.MXSerializer" );
117         jClass.addImport( "java.io.OutputStream" );
118         jClass.addImport( "java.io.Writer" );
119         jClass.addImport( "java.util.Iterator" );
120 
121         JField namespaceField = new JField( new JClass( "String" ), "NAMESPACE" );
122         namespaceField.getModifiers().setFinal( true );
123         namespaceField.getModifiers().setStatic( true );
124         namespaceField.setInitString( "null" );
125         jClass.addField( namespaceField );
126 
127         JField commentField = new JField( new JClass( "String" ), "fileComment" );
128         commentField.setInitString( "null" );
129         jClass.addField( commentField );
130 
131         // Add setComment method
132         JMethod setComment = new JMethod( "setFileComment" );
133 
134         setComment.addParameter( new JParameter( new JClass( "String" ), "fileComment" ) );
135         JSourceCode setCommentSourceCode = setComment.getSourceCode();
136         setCommentSourceCode.add( "this.fileComment = fileComment;" );
137         jClass.addMethod( setComment );
138 
139         addModelImports( jClass, null );
140 
141         String root = objectModel.getRoot( getGeneratedVersion() );
142 
143         ModelClass rootClass = objectModel.getClass( root, getGeneratedVersion() );
144 
145         String rootElement = resolveTagName( rootClass );
146 
147         // ----------------------------------------------------------------------
148         // Write the write( Writer, Model ) method which will do the unmarshalling.
149         // ----------------------------------------------------------------------
150 
151         JMethod marshall = new JMethod( "write" );
152 
153         String rootElementParameterName = uncapitalise( root );
154         marshall.addParameter( new JParameter( new JClass( "Writer" ), "writer" ) );
155         marshall.addParameter( new JParameter( new JClass( root ), rootElementParameterName ) );
156 
157         marshall.addException( new JClass( "java.io.IOException" ) );
158 
159         JSourceCode sc = marshall.getSourceCode();
160 
161         sc.add( "XmlSerializer serializer = new MXSerializer();" );
162 
163         sc.add(
164             "serializer.setProperty( \"http://xmlpull.org/v1/doc/properties.html#serializer-indentation\", \"  \" );" );
165 
166         sc.add(
167             "serializer.setProperty( \"http://xmlpull.org/v1/doc/properties.html#serializer-line-separator\", \"\\n\" );" );
168 
169         sc.add( "serializer.setOutput( writer );" );
170 
171         sc.add( "serializer.startDocument( " + rootElementParameterName + ".getModelEncoding(), null );" );
172 
173         sc.add( "write" + root + "( " + rootElementParameterName + ", \"" + rootElement + "\", serializer );" );
174 
175         sc.add( "serializer.endDocument();" );
176 
177         jClass.addMethod( marshall );
178 
179         // ----------------------------------------------------------------------
180         // Write the write( OutputStream, Model ) method which will do the unmarshalling.
181         // ----------------------------------------------------------------------
182 
183         marshall = new JMethod( "write" );
184 
185         marshall.addParameter( new JParameter( new JClass( "OutputStream" ), "stream" ) );
186         marshall.addParameter( new JParameter( new JClass( root ), rootElementParameterName ) );
187 
188         marshall.addException( new JClass( "java.io.IOException" ) );
189 
190         sc = marshall.getSourceCode();
191 
192         sc.add( "XmlSerializer serializer = new MXSerializer();" );
193 
194         sc.add(
195             "serializer.setProperty( \"http://xmlpull.org/v1/doc/properties.html#serializer-indentation\", \"  \" );" );
196 
197         sc.add(
198             "serializer.setProperty( \"http://xmlpull.org/v1/doc/properties.html#serializer-line-separator\", \"\\n\" );" );
199 
200         sc.add( "serializer.setOutput( stream, " + rootElementParameterName + ".getModelEncoding() );" );
201 
202         sc.add( "serializer.startDocument( " + rootElementParameterName + ".getModelEncoding(), null );" );
203 
204         sc.add( "write" + root + "( " + rootElementParameterName + ", \"" + rootElement + "\", serializer );" );
205 
206         sc.add( "serializer.endDocument();" );
207 
208         jClass.addMethod( marshall );
209 
210         writeAllClasses( objectModel, jClass );
211 
212         if ( isLocationTracking() )
213         {
214             prepareLocationTracking( jClass );
215         }
216 
217         if ( requiresDomSupport )
218         {
219             createWriteDomMethod( jClass );
220         }
221 
222         jClass.print( sourceWriter );
223 
224         sourceWriter.close();
225     }

227     private void writeAllClasses( Model objectModel, JClass jClass )
228         throws ModelloException
229     {
230         for ( ModelClass clazz : getClasses( objectModel ) )
231         {
232             writeClass( clazz, jClass );
233         }
234     }

236     private void writeClass( ModelClass modelClass, JClass jClass )
237         throws ModelloException
238     {
239         String className = modelClass.getName();
240 
241         String uncapClassName = uncapitalise( className );
242 
243         JMethod marshall = new JMethod( "write" + className );
244 
245         marshall.addParameter( new JParameter( new JClass( className ), uncapClassName ) );
246         marshall.addParameter( new JParameter( new JClass( "String" ), "tagName" ) );
247         marshall.addParameter( new JParameter( new JClass( "XmlSerializer" ), "serializer" ) );
248 
249         marshall.addException( new JClass( "java.io.IOException" ) );
250 
251         marshall.getModifiers().makePrivate();
252 
253         JSourceCode sc = marshall.getSourceCode();
254 
255         ModelClassMetadata classMetadata = (ModelClassMetadata) modelClass.getMetadata( ModelClassMetadata.ID );
256 
257         String namespace = null;
258         XmlModelMetadata xmlModelMetadata = (XmlModelMetadata) modelClass.getModel().getMetadata( XmlModelMetadata.ID );
259 
260         // add namespace information for root element only
261         if ( classMetadata.isRootElement() && ( xmlModelMetadata.getNamespace() != null ) )
262         {
263             sc.add( "if ( this.fileComment != null )" );
264             sc.add( "{" );
265             sc.add( "serializer.comment(this.fileComment);" );
266             sc.add( "}" );
267 
268             namespace = xmlModelMetadata.getNamespace( getGeneratedVersion() );
269             sc.add( "serializer.setPrefix( \"\", \"" + namespace + "\" );" );
270         }
271 
272         if ( ( namespace != null ) && ( xmlModelMetadata.getSchemaLocation() != null ) )
273         {
274             String url = xmlModelMetadata.getSchemaLocation( getGeneratedVersion() );
275 
276             sc.add( "serializer.setPrefix( \"xsi\", \"http://www.w3.org/2001/XMLSchema-instance\" );" );
277 
278             sc.add( "serializer.startTag( NAMESPACE, tagName );" );
279 
280             sc.add( "serializer.attribute( \"\", \"xsi:schemaLocation\", \"" + namespace + " " + url + "\" );" );
281         }
282         else
283         {
284             sc.add( "serializer.startTag( NAMESPACE, tagName );" );
285         }
286 
287         ModelField contentField = null;
288 
289         String contentValue = null;
290 
291         List<ModelField> modelFields = getFieldsForXml( modelClass, getGeneratedVersion() );
292 
293         // XML attributes
294         for ( ModelField field : modelFields )
295         {
296             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
297 
298             JavaFieldMetadata javaFieldMetadata = (JavaFieldMetadata) field.getMetadata( JavaFieldMetadata.ID );
299 
300             String fieldTagName = resolveTagName( field, xmlFieldMetadata );
301 
302             String type = field.getType();
303 
304             String value = uncapClassName + "." + getPrefix( javaFieldMetadata ) + capitalise( field.getName() ) + "()";
305 
306             if ( xmlFieldMetadata.isContent() )
307             {
308                 contentField = field;
309                 contentValue = value;
310                 continue;
311             }
312 
313             if ( xmlFieldMetadata.isAttribute() )
314             {
315                 sc.add( getValueChecker( type, value, field ) );
316 
317                 sc.add( "{" );
318                 sc.addIndented( "serializer.attribute( NAMESPACE, \"" + fieldTagName + "\", "
319                                 + getValue( field.getType(), value, xmlFieldMetadata ) + " );" );
320                 sc.add( "}" );
321             }
322 
323         }
324 
325         if ( contentField != null )
326         {
327             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) contentField.getMetadata( XmlFieldMetadata.ID );
328             sc.add( "serializer.text( " + getValue( contentField.getType(), contentValue, xmlFieldMetadata ) + " );" );
329         }
330 
331         // XML tags
332         for ( ModelField field : modelFields )
333         {
334             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
335 
336             if ( xmlFieldMetadata.isContent() )
337             {
338                 // skip field with type Content
339                 continue;
340             }
341 
342             JavaFieldMetadata javaFieldMetadata = (JavaFieldMetadata) field.getMetadata( JavaFieldMetadata.ID );
343 
344             String fieldTagName = resolveTagName( field, xmlFieldMetadata );
345 
346             String type = field.getType();
347 
348             String value = uncapClassName + "." + getPrefix( javaFieldMetadata ) + capitalise( field.getName() ) + "()";
349 
350             if ( xmlFieldMetadata.isAttribute() )
351             {
352                 continue;
353             }
354 
355             if ( field instanceof ModelAssociation )
356             {
357                 ModelAssociation association = (ModelAssociation) field;
358 
359                 String associationName = association.getName();
360 
361                 if ( association.isOneMultiplicity() )
362                 {
363                     sc.add( getValueChecker( type, value, association ) );
364 
365                     sc.add( "{" );
366                     sc.addIndented( "write" + association.getTo() + "( (" + association.getTo() + ") " + value + ", \""
367                                     + fieldTagName + "\", serializer );" );
368                     sc.add( "}" );
369                 }
370                 else
371                 {
372                     //MANY_MULTIPLICITY
373 
374                     XmlAssociationMetadata xmlAssociationMetadata =
375                         (XmlAssociationMetadata) association.getAssociationMetadata( XmlAssociationMetadata.ID );
376 
377                     String valuesTagName = resolveTagName( fieldTagName, xmlAssociationMetadata );
378 
379                     type = association.getType();
380                     String toType = association.getTo();
381 
382                     boolean wrappedItems = xmlAssociationMetadata.isWrappedItems();
383 
384                     if ( ModelDefault.LIST.equals( type ) || ModelDefault.SET.equals( type ) )
385                     {
386                         boolean isList = ModelDefault.LIST.equals( type );
387 
388                         sc.add( getValueChecker( type, value, association ) );
389 
390                         sc.add( "{" );
391                         sc.indent();
392 
393                         if ( wrappedItems )
394                         {
395                             sc.add( "serializer.startTag( NAMESPACE, " + "\"" + fieldTagName + "\" );" );
396                         }
397 
398                         if ( isLocationTracking() && !isClassInModel( association.getTo(), modelClass.getModel() ) )
399                         {
400                             sc.add( locationTracker.getName() + " location = " + uncapClassName + ".getLocation( \"" + fieldTagName + "\" );" );
401                             if ( isList )
402                             {
403                                 sc.add( "int n = 0;" );
404                             }
405                         }
406 
407                         sc.add( "for ( Iterator iter = " + value + ".iterator(); iter.hasNext(); )" );
408 
409                         sc.add( "{" );
410                         sc.indent();
411 
412                         if ( isClassInModel( association.getTo(), modelClass.getModel() ) )
413                         {
414                             sc.add( toType + " o = (" + toType + ") iter.next();" );
415 
416                             sc.add( "write" + toType + "( o, \"" + valuesTagName + "\", serializer );" );
417                         }
418                         else
419                         {
420                             String variable = singular( uncapitalise( field.getName() ) );
421 
422                             sc.add( toType + " " + variable + " = (" + toType + ") iter.next();" );
423 
424                             sc.add( "serializer.startTag( NAMESPACE, \"" + valuesTagName + "\" ).text( " + variable
425                                 + " ).endTag( NAMESPACE, \"" + valuesTagName + "\" );" );
426 
427                             writeLocationTracking( sc, "location", isList ? "Integer.valueOf( n++ )" : variable );
428                         }
429 
430                         sc.unindent();
431                         sc.add( "}" );
432 
433                         if ( wrappedItems )
434                         {
435                             sc.add( "serializer.endTag( NAMESPACE, \"" + fieldTagName + "\" );" );
436                         }
437 
438                         sc.unindent();
439                         sc.add( "}" );
440                     }
441                     else
442                     {
443                         //Map or Properties
444 
445                         sc.add( getValueChecker( type, value, field ) );
446 
447                         sc.add( "{" );
448                         sc.indent();
449 
450                         if ( wrappedItems )
451                         {
452                             sc.add( "serializer.startTag( NAMESPACE, \"" + fieldTagName + "\" );" );
453                         }
454 
455                         if ( isLocationTracking() )
456                         {
457                             sc.add( locationTracker.getName() + " location = " + uncapClassName + ".getLocation( \"" + fieldTagName + "\" );" );
458                         }
459 
460                         sc.add( "for ( Iterator iter = " + value + ".keySet().iterator(); iter.hasNext(); )" );
461 
462                         sc.add( "{" );
463                         sc.indent();
464 
465                         sc.add( "String key = (String) iter.next();" );
466 
467                         sc.add( "String value = (String) " + value + ".get( key );" );
468 
469                         if ( xmlAssociationMetadata.isMapExplode() )
470                         {
471                             sc.add( "serializer.startTag( NAMESPACE, \"" + singular( associationName ) + "\" );" );
472                             sc.add(
473                                 "serializer.startTag( NAMESPACE, \"key\" ).text( key ).endTag( NAMESPACE, \"key\" );" );
474                             sc.add(
475                                 "serializer.startTag( NAMESPACE, \"value\" ).text( value ).endTag( NAMESPACE, \"value\" );" );
476                             sc.add( "serializer.endTag( NAMESPACE, \"" + singular( associationName ) + "\" );" );
477                         }
478                         else
479                         {
480                             sc.add(
481                                 "serializer.startTag( NAMESPACE, key ).text( value ).endTag( NAMESPACE, key );" );
482                         }
483 
484                         writeLocationTracking( sc, "location", "key" );
485 
486                         sc.unindent();
487                         sc.add( "}" );
488 
489                         if ( wrappedItems )
490                         {
491                             sc.add( "serializer.endTag( NAMESPACE, \"" + fieldTagName + "\" );" );
492                         }
493 
494                         sc.unindent();
495                         sc.add( "}" );
496                     }
497                 }
498             }
499             else
500             {
501                 sc.add( getValueChecker( type, value, field ) );
502 
503                 sc.add( "{" );
504                 if ( "DOM".equals( field.getType() ) )
505                 {
506                     if ( domAsXpp3 )
507                     {
508                         jClass.addImport( "org.codehaus.plexus.util.xml.Xpp3Dom" );
509 
510                         if ( isLocationTracking() )
511                         {
512                             sc.addIndented( "writeXpp3DomToSerializer( (Xpp3Dom) " + value + ", serializer );" );
513                         }
514                         else
515                         {
516                             sc.addIndented( "((Xpp3Dom) " + value + ").writeToSerializer( NAMESPACE, serializer );" );
517                         }
518                     }
519                     else
520                     {
521                         sc.addIndented( "writeDom( (org.w3c.dom.Element) " + value + ", serializer );" );
522                     }
523 
524                     requiresDomSupport = true;
525                 }
526                 else
527                 {
528                     sc.addIndented( "serializer.startTag( NAMESPACE, " + "\"" + fieldTagName + "\" ).text( "
529                         + getValue( field.getType(), value, xmlFieldMetadata ) + " ).endTag( NAMESPACE, " + "\""
530                         + fieldTagName + "\" );" );
531                     sc.indent();
532                     writeLocationTracking( sc, uncapClassName, '"' + fieldTagName + '"' );
533                     sc.unindent();
534                 }
535                 sc.add( "}" );
536             }
537         }
538 
539         sc.add( "serializer.endTag( NAMESPACE, tagName );" );
540 
541         jClass.addMethod( marshall );
542     }

544     private void createWriteDomMethod( JClass jClass )
545     {
546         if ( domAsXpp3 )
547         {
548             return;
549         }
550         String type = "org.w3c.dom.Element";
551         JMethod method = new JMethod( "writeDom" );
552         method.getModifiers().makePrivate();
553 
554         method.addParameter( new JParameter( new JType( type ), "dom" ) );
555         method.addParameter( new JParameter( new JClass( "XmlSerializer" ), "serializer" ) );
556 
557         method.addException( new JClass( "java.io.IOException" ) );
558 
559         JSourceCode sc = method.getSourceCode();
560 
561         // start element
562         sc.add( "serializer.startTag( NAMESPACE, dom.getTagName() );" );
563 
564         // attributes
565         sc.add( "org.w3c.dom.NamedNodeMap attributes = dom.getAttributes();" );
566         sc.add( "for ( int i = 0; i < attributes.getLength(); i++ )" );
567         sc.add( "{" );
568 
569         sc.indent();
570         sc.add( "org.w3c.dom.Node attribute = attributes.item( i );" );
571         sc.add( "serializer.attribute( NAMESPACE, attribute.getNodeName(), attribute.getNodeValue() );" );
572         sc.unindent();
573 
574         sc.add( "}" );
575 
576         // child nodes & text
577         sc.add( "org.w3c.dom.NodeList children = dom.getChildNodes();" );
578         sc.add( "for ( int i = 0; i < children.getLength(); i++ )" );
579         sc.add( "{" );
580         sc.indent();
581         sc.add( "org.w3c.dom.Node node = children.item( i );" );
582         sc.add( "if ( node instanceof org.w3c.dom.Element)" );
583         sc.add( "{" );
584         sc.addIndented( "writeDom( (org.w3c.dom.Element) children.item( i ), serializer );" );
585         sc.add( "}" );
586         sc.add( "else" );
587         sc.add( "{" );
588         sc.addIndented( "serializer.text( node.getTextContent() );" );
589         sc.add( "}" );
590         sc.unindent();
591         sc.add( "}" );
592 
593         sc.add( "serializer.endTag( NAMESPACE, dom.getTagName() );" );
594 
595         jClass.addMethod( method );
596     }
