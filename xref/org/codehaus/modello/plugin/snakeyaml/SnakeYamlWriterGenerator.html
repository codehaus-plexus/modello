53      public void generate( Model model, Properties parameters )
54          throws ModelloException
55      {
56          initialize( model, parameters );
57  
58          try
59          {
60              generateSnakeYamlWriter();
61          }
62          catch ( IOException ex )
63          {
64              throw new ModelloException( "Exception while generating SnakeYaml Writer.", ex );
65          }
66      }
67  
68      private void generateSnakeYamlWriter()
69          throws ModelloException, IOException
70      {
71          Model objectModel = getModel();
72  
73          String packageName = objectModel.getDefaultPackageName( isPackageWithVersion(), getGeneratedVersion() )
74              + ".io.snakeyaml";
75  
76          String marshallerName = getFileName( "SnakeYamlWriter" );
77  
78          JSourceWriter sourceWriter = newJSourceWriter( packageName, marshallerName );
79  
80          JClass jClass = new JClass( packageName + '.' + marshallerName );
81          initHeader( jClass );
82  
83          jClass.addImport( "org.yaml.snakeyaml.DumperOptions" );
84          jClass.addImport( "org.yaml.snakeyaml.DumperOptions.Version" );
85          jClass.addImport( "org.yaml.snakeyaml.DumperOptions.FlowStyle" );
86          jClass.addImport( "org.yaml.snakeyaml.DumperOptions.ScalarStyle" );
87          jClass.addImport( "org.yaml.snakeyaml.emitter.Emitable" );
88          jClass.addImport( "org.yaml.snakeyaml.emitter.Emitter" );
89          jClass.addImport( "org.yaml.snakeyaml.error.Mark" );
90          jClass.addImport( "org.yaml.snakeyaml.events.DocumentEndEvent" );
91          jClass.addImport( "org.yaml.snakeyaml.events.DocumentStartEvent" );
92          jClass.addImport( "org.yaml.snakeyaml.events.ImplicitTuple" );
93          jClass.addImport( "org.yaml.snakeyaml.events.MappingEndEvent" );
94          jClass.addImport( "org.yaml.snakeyaml.events.MappingStartEvent" );
95          jClass.addImport( "org.yaml.snakeyaml.events.ScalarEvent" );
96          jClass.addImport( "org.yaml.snakeyaml.events.SequenceEndEvent" );
97          jClass.addImport( "org.yaml.snakeyaml.events.SequenceStartEvent" );
98          jClass.addImport( "org.yaml.snakeyaml.events.StreamEndEvent" );
99          jClass.addImport( "org.yaml.snakeyaml.events.StreamStartEvent" );
100         jClass.addImport( "java.io.IOException" );
101         jClass.addImport( "java.io.OutputStream" );
102         jClass.addImport( "java.io.OutputStreamWriter" );
103         jClass.addImport( "java.io.Writer" );
104 
105         addModelImports( jClass, null );
106 
107         JField factoryField = new JField( new JClass( "DumperOptions" ), "dumperOptions" );
108         factoryField.getModifiers().setFinal( true );
109         factoryField.setInitString( "new DumperOptions()" );
110         jClass.addField( factoryField );
111 
112         JConstructor jacksonWriterConstructor = new JConstructor( jClass );
113         JSourceCode sc = jacksonWriterConstructor.getSourceCode();
114         sc.add( "dumperOptions.setAllowUnicode( true );" );
115         sc.add( "dumperOptions.setPrettyFlow( true );" );
116         sc.add( "dumperOptions.setVersion( Version.V1_1 );" );
117 
118         jClass.addConstructor( jacksonWriterConstructor );
119 
120         String root = objectModel.getRoot( getGeneratedVersion() );
121 
122         // ----------------------------------------------------------------------
123         // Write the write( Writer, Model ) method which will do the unmarshalling.
124         // ----------------------------------------------------------------------
125 
126         JMethod marshall = new JMethod( "write" );
127 
128         String rootElementParameterName = uncapitalise( root );
129         marshall.addParameter( new JParameter( new JClass( "Writer" ), "writer" ) );
130         marshall.addParameter( new JParameter( new JClass( root ), rootElementParameterName ) );
131 
132         marshall.addException( new JClass( "IOException" ) );
133 
134         sc = marshall.getSourceCode();
135 
136         sc.add( "Emitable generator = new Emitter( writer, dumperOptions );" );
137 
138         sc.add( "generator.emit( new StreamStartEvent( null, null ) );" );
139 
140         sc.add( "generator.emit( new DocumentStartEvent( null, null, dumperOptions.isExplicitStart(), dumperOptions.getVersion(), dumperOptions.getTags() ) );" );
141 
142         sc.add( "write" + root + "( " + rootElementParameterName + ", generator );" );
143 
144         sc.add( "generator.emit( new DocumentEndEvent( null, null, dumperOptions.isExplicitEnd() ) );" );
145 
146         sc.add( "generator.emit( new StreamEndEvent( null, null ) );" );
147 
148         jClass.addMethod( marshall );
149 
150         // ----------------------------------------------------------------------
151         // Write the write( OutputStream, Model ) method which will do the unmarshalling.
152         // ----------------------------------------------------------------------
153 
154         marshall = new JMethod( "write" );
155 
156         marshall.addParameter( new JParameter( new JClass( "OutputStream" ), "stream" ) );
157         marshall.addParameter( new JParameter( new JClass( root ), rootElementParameterName ) );
158 
159         marshall.addException( new JClass( "IOException" ) );
160 
161         sc = marshall.getSourceCode();
162 
163         sc.add( "write( new OutputStreamWriter( stream, "
164                         + rootElementParameterName
165                         + ".getModelEncoding() ), "
166                         + rootElementParameterName
167                         + " );" );
168 
169         jClass.addMethod( marshall );
170 
171         writeAllClasses( objectModel, jClass );
172 
173         jClass.print( sourceWriter );
174 
175         sourceWriter.close();
176     }
177 
178     private void writeAllClasses( Model objectModel, JClass jClass )
179         throws ModelloException
180     {
181         for ( ModelClass clazz : getClasses( objectModel ) )
182         {
183             writeClass( clazz, jClass );
184         }
185     }
186 
187     private void writeClass( ModelClass modelClass, JClass jClass )
188         throws ModelloException
189     {
190         String className = modelClass.getName();
191 
192         String uncapClassName = uncapitalise( className );
193 
194         JMethod marshall = new JMethod( "write" + className );
195 
196         marshall.addParameter( new JParameter( new JClass( className ), uncapClassName ) );
197         marshall.addParameter( new JParameter( new JClass( "Emitable" ), "generator" ) );
198 
199         marshall.addException( new JClass( "IOException" ) );
200 
201         marshall.getModifiers().makePrivate();
202 
203         JSourceCode sc = marshall.getSourceCode();
204 
205         sc.add( "generator.emit( new MappingStartEvent( null, null, true, null, null, FlowStyle.BLOCK ) );" );
206 
207         ModelField contentField = null;
208 
209         String contentValue = null;
210 
211         List<ModelField> modelFields = getFieldsForXml( modelClass, getGeneratedVersion() );
212 
213         // XML attributes
214         for ( ModelField field : modelFields )
215         {
216             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
217 
218             JavaFieldMetadata javaFieldMetadata = (JavaFieldMetadata) field.getMetadata( JavaFieldMetadata.ID );
219 
220             String fieldTagName = resolveTagName( field, xmlFieldMetadata );
221 
222             String type = field.getType();
223 
224             String value = uncapClassName + "." + getPrefix( javaFieldMetadata ) + capitalise( field.getName() ) + "()";
225 
226             if ( xmlFieldMetadata.isContent() )
227             {
228                 contentField = field;
229                 contentValue = value;
230                 continue;
231             }
232 
233             if ( xmlFieldMetadata.isAttribute() )
234             {
235                 sc.add( getValueChecker( type, value, field ) );
236 
237                 sc.add( "{" );
238                 sc.indent();
239 
240                 writeScalarKey( sc, fieldTagName );
241                 writeScalar( sc, getValue( field.getType(), value, xmlFieldMetadata ) );
242 
243                 sc.unindent();
244                 sc.add( "}" );
245             }
246 
247         }
248 
249         if ( contentField != null )
250         {
251             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) contentField.getMetadata( XmlFieldMetadata.ID );
252             writeScalar( sc, getValue( contentField.getType(), contentValue, xmlFieldMetadata ) );
253         }
254 
255         final boolean useJava5 = hasJavaSourceSupport( 5 );
256 
257         // XML tags
258         for ( ModelField field : modelFields )
259         {
260             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
261 
262             if ( xmlFieldMetadata.isContent() )
263             {
264                 // skip field with type Content
265                 continue;
266             }
267 
268             JavaFieldMetadata javaFieldMetadata = (JavaFieldMetadata) field.getMetadata( JavaFieldMetadata.ID );
269 
270             String fieldTagName = resolveTagName( field, xmlFieldMetadata );
271 
272             String type = field.getType();
273 
274             String value = uncapClassName + "." + getPrefix( javaFieldMetadata ) + capitalise( field.getName() ) + "()";
275 
276             if ( xmlFieldMetadata.isAttribute() )
277             {
278                 continue;
279             }
280 
281             if ( field instanceof ModelAssociation )
282             {
283                 ModelAssociation association = (ModelAssociation) field;
284 
285                 if ( association.isOneMultiplicity() )
286                 {
287                     sc.add( getValueChecker( type, value, association ) );
288 
289                     sc.add( "{" );
290                     sc.indent();
291 
292                     writeScalarKey( sc, fieldTagName );
293                     sc.add( "write" + association.getTo() + "( (" + association.getTo() + ") " + value + ", generator );" );
294 
295                     sc.unindent();
296                     sc.add( "}" );
297                 }
298                 else
299                 {
300                     //MANY_MULTIPLICITY
301 
302                     XmlAssociationMetadata xmlAssociationMetadata =
303                         (XmlAssociationMetadata) association.getAssociationMetadata( XmlAssociationMetadata.ID );
304 
305                     type = association.getType();
306                     String toType = association.getTo();
307 
308                     if ( ModelDefault.LIST.equals( type ) || ModelDefault.SET.equals( type ) )
309                     {
310                         sc.add( getValueChecker( type, value, association ) );
311 
312                         sc.add( "{" );
313                         sc.indent();
314 
315                         writeScalarKey( sc, fieldTagName );
316                         sc.add( "generator.emit( new SequenceStartEvent( null, null, true, null, null, false ) );" );
317 
318                         if ( useJava5 )
319                         {
320                             sc.add( "for ( " + toType + " o : " + value + " )" );
321                         }
322                         else
323                         {
324                             sc.add( "for ( java.util.Iterator it = " + value + ".iterator(); it.hasNext(); )" );
325                         }
326 
327                         sc.add( "{" );
328                         sc.indent();
329 
330                         if ( !useJava5 )
331                         {
332                             sc.add( toType + " o = (" + toType + " ) it.next();" );
333                         }
334 
335                         if ( isClassInModel( association.getTo(), modelClass.getModel() ) )
336                         {
337                             sc.add( "write" + toType + "( o, generator );" );
338                         }
339                         else
340                         {
341                             writeScalar( sc, "o" );
342                         }
343 
344                         sc.unindent();
345                         sc.add( "}" );
346 
347                         sc.add( "generator.emit( new SequenceEndEvent( null, null ) );" );
348 
349                         sc.unindent();
350                         sc.add( "}" );
351                     }
352                     else
353                     {
354                         //Map or Properties
355 
356                         sc.add( getValueChecker( type, value, field ) );
357 
358                         sc.add( "{" );
359                         sc.indent();
360 
361                         writeScalarKey( sc, fieldTagName );
362 
363                         if ( xmlAssociationMetadata.isMapExplode() )
364                         {
365                             sc.add( "generator.emit( new SequenceStartEvent( null, null, true, null, null, false ) );" );
366                         }
367                         else
368                         {
369                             sc.add( "generator.emit( new MappingStartEvent( null, null, true, null, null, FlowStyle.BLOCK ) );" );
370                         }
371 
372 
373 
374                         StringBuilder entryTypeBuilder = new StringBuilder( "java.util.Map.Entry" );
375 
376                         if ( useJava5 )
377                         {
378                             entryTypeBuilder.append( '<' );
379 
380                             if ( association.getType().equals( ModelDefault.PROPERTIES ) )
381                             {
382                                 entryTypeBuilder.append( "Object, Object" );
383                             }
384                             else
385                             {
386                                 entryTypeBuilder.append( "String, " ).append( association.getTo() );
387                             }
388 
389                             entryTypeBuilder.append( '>' );
390                         }
391 
392                         if ( useJava5 )
393                         {
394                             sc.add( "for ( " + entryTypeBuilder + " entry : " + value + ".entrySet() )" );
395                         }
396                         else
397                         {
398                             sc.add( "for ( java.util.Iterator it = " + value + ".entrySet().iterator(); it.hasNext(); )" );
399                         }
400 
401                         sc.add( "{" );
402                         sc.indent();
403 
404                         if ( !useJava5 )
405                         {
406                             sc.add( entryTypeBuilder + " entry = (" + entryTypeBuilder + ") it.next();" );
407                         }
408 
409                         sc.add( "final String key = String.valueOf( entry.getKey() );" );
410                         sc.add( "final String value = String.valueOf( entry.getValue() );" );
411 
412                         if ( xmlAssociationMetadata.isMapExplode() )
413                         {
414                             sc.add( "generator.emit( new MappingStartEvent( null, null, true, null, null, FlowStyle.BLOCK) );" );
415                             writeScalarKey( sc, "key" );
416                             writeScalar( sc, "key" );
417                             writeScalarKey( sc, "value" );
418                             writeScalar( sc, "value" );
419                             sc.add( "generator.emit( new MappingEndEvent( null, null ) );" );
420                         }
421                         else
422                         {
423                             writeScalar( sc, "key" );
424                             writeScalar( sc, "value" );
425                         }
426 
427                         sc.unindent();
428                         sc.add( "}" );
429 
430                         if ( xmlAssociationMetadata.isMapExplode() )
431                         {
432                             sc.add( "generator.emit( new SequenceEndEvent( null, null ) );" );
433                         }
434                         else
435                         {
436                             sc.add( "generator.emit( new MappingEndEvent( null, null ) );" );
437                         }
438 
439                         sc.unindent();
440                         sc.add( "}" );
441                     }
442                 }
443             }
444             else
445             {
446                 sc.add( getValueChecker( type, value, field ) );
447 
448                 sc.add( "{" );
449                 sc.indent();
450 
451                 writeScalarKey( sc, fieldTagName );
452                 writeScalar( sc, getValue( field.getType(), value, xmlFieldMetadata ) );
453 
454                 sc.unindent();
455                 sc.add( "}" );
456             }
457         }
458 
459         sc.add( "generator.emit( new MappingEndEvent( null, null ) );" );
460 
461         jClass.addMethod( marshall );
462     }
463 
464     private void writeScalarKey( JSourceCode sc, String key )
465     {
466         writeScalar( sc, "\"" + key + "\"" );
467     }
468 
469     private void writeScalar( JSourceCode sc, String value )
470     {
471         sc.add( "{" );
472         sc.indent();
473         sc.add( "String anchor = null, tag = null;" );
474         sc.add( "Mark startMark = null, endMark = null;" );
475         sc.add( "ScalarStyle style = ScalarStyle.DOUBLE_QUOTED;" );
476         sc.add( "generator.emit( new ScalarEvent( anchor, tag, new ImplicitTuple( true, true ), "
477                 + value
478                 + ", startMark, endMark, style ) );" );
479         sc.unindent();
480         sc.add( "}" );
481     }
