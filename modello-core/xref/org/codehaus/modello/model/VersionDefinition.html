25  public class VersionDefinition {
26      private static final String FIELD = "field";
27  
28      private static final String NAMESPACE = "namespace";
29  
30      private static final String FIELD_NAMESPACE = "field+namespace";
31  
32      private String type;
33  
34      private String value;
35  
36      public String getType() {
37          return type;
38      }
39  
40      public void setType(String type) {
41          this.type = type;
42      }
43  
44      public String getValue() {
45          return value;
46      }
47  
48      public void setValue(String value) {
49          this.value = value;
50      }
51  
52      /**
53       *
54       * @return {@code true} if the versionDefinition can be based on the namespace
55       * @since 1.9
56       */
57      public boolean isNamespaceType() {
58          return NAMESPACE.equals(type) || FIELD_NAMESPACE.equals(type);
59      }
60  
61      /**
62       *
63       * @return {@code true} if the versionDefinition can be based on the field
64       * @since 1.9
65       */
66      public boolean isFieldType() {
67          return FIELD.equals(type) || FIELD_NAMESPACE.equals(type);
68      }
69  }
