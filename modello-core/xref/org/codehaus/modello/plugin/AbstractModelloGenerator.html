59  public abstract class AbstractModelloGenerator implements ModelloGenerator {
60      private final Logger logger = LoggerFactory.getLogger(getClass());
61  
62      private static final Map<String, String> PLURAL_EXCEPTIONS = new HashMap<>();
63  
64      private Model model;
65  
66      private File outputDirectory;
67  
68      private Version generatedVersion;
69  
70      private boolean packageWithVersion;
71  
72      private String encoding;
73  
74      private List<String> licenseText;
75  
76      @Inject
77      private BuildContext buildContext;
78  
79      protected Logger getLogger() {
80          return logger;
81      }
82  
83      @SuppressWarnings("uncheked")
84      protected void initialize(Model model, Map<String, Object> parameters) throws ModelloException {
85          this.model = model;
86  
87          outputDirectory = new File(requireParameter(parameters, ModelloParameterConstants.OUTPUT_DIRECTORY));
88  
89          String version = requireParameter(parameters, ModelloParameterConstants.VERSION);
90  
91          generatedVersion = new Version(version);
92  
93          packageWithVersion =
94                  Boolean.parseBoolean(requireParameter(parameters, ModelloParameterConstants.PACKAGE_WITH_VERSION));
95  
96          encoding = (String) parameters.get(ModelloParameterConstants.ENCODING);
97  
98          licenseText = (List<String>) parameters.get(ModelloParameterConstants.LICENSE_TEXT);
99  
100         Optional.ofNullable(parameters.get(ModelloParameterConstants.PLURAL_EXCEPTIONS))
101                 .ifPresent(o -> PLURAL_EXCEPTIONS.putAll((Map<String, String>) o));
102     }
103

124     protected List<String> getHeader() {
125         List<String> header = new ArrayList<>();
126         List<String> license = getLicenseHeader();
127         if (license != null) {
128             header.addAll(license);
129         }
130         List<String> generated = getGeneratedHeader();
131         if (generated != null) {
132             header.addAll(generated);
133         }
134         return header;
135     }
136 
137     protected List<String> getGeneratedHeader() {
138         String version = getClass().getPackage().getImplementationVersion();
139         return Arrays.asList(
140                 "=================== DO NOT EDIT THIS FILE ====================",
141                 "Generated by Modello" + ((version == null) ? "" : (' ' + version)) + ",",
142                 "any modifications will be overwritten.",
143                 "==============================================================");
144     }
145 
146     protected List<String> getLicenseHeader() {
147         return licenseText;
148     }
149 
150     protected boolean isClassInModel(String fieldType, Model model) {
151         try {
152             return model.getClass(fieldType, generatedVersion) != null;
153         } catch (Exception e) {
154         }
155 
156         return false;
157     }
158 
159     /**
160      * Return the child fields of this class.
161      *
162      * @param modelClass current class
163      * @return the list of fields of this class
164      */
165     protected List<ModelField> getFieldsForClass(ModelClass modelClass) {
166         List<ModelField> fields = new ArrayList<ModelField>();
167 
168         while (modelClass != null) {
169             fields.addAll(modelClass.getFields(getGeneratedVersion()));
170 
171             String superClass = modelClass.getSuperClass();
172             if (superClass != null) {
173                 modelClass = getModel().getClass(superClass, getGeneratedVersion());
174             } else {
175                 modelClass = null;
176             }
177         }
178 
179         return fields;
180     }
181 
182     protected boolean isInnerAssociation(ModelField field) {
183         return field instanceof ModelAssociation && isClassInModel(((ModelAssociation) field).getTo(), getModel());
184     }

186     protected boolean isMap(String fieldType) {
187         return ModelDefault.MAP.equals(fieldType) || ModelDefault.PROPERTIES.equals(fieldType);
188     }
189 
190     protected boolean isCollection(String fieldType) {
191         return ModelDefault.LIST.equals(fieldType) || ModelDefault.SET.equals(fieldType);
192     }

194     protected String capitalise(String str) {
195         if (StringUtils.isEmpty(str)) {
196             return str;
197         }
198 
199         return new StringBuilder(str.length())
200                 .append(Character.toTitleCase(str.charAt(0)))
201                 .append(str.substring(1))
202                 .toString();
203     }
204 
205     public static String singular(String name) {
206         if (StringUtils.isEmpty(name)) {
207             return name;
208         }
209 
210         if (PLURAL_EXCEPTIONS.containsKey(name)) {
211             return PLURAL_EXCEPTIONS.get(name);
212         }
213 
214         if (name.endsWith("ies")) {
215             return name.substring(0, name.length() - 3) + "y";
216         } else if (name.endsWith("es") && name.endsWith("ches")) {
217             return name.substring(0, name.length() - 2);
218         } else if (name.endsWith("xes")) {
219             return name.substring(0, name.length() - 2);
220         } else if (name.endsWith("s") && (name.length() != 1)) {
221             return name.substring(0, name.length() - 1);
222         }
223 
224         return name;
225     }
226 
227     public static String uncapitalise(String str) {
228         if (StringUtils.isEmpty(str)) {
229             return str;
230         }
231 
232         return new StringBuilder(str.length())
233                 .append(Character.toLowerCase(str.charAt(0)))
234                 .append(str.substring(1))
235                 .toString();
236     }

250     protected String requireParameter(Map<String, Object> parameters, String name) {
251         String value = (String) parameters.get(name);
252 
253         if (value == null) {
254             throw new ModelloRuntimeException("Missing parameter '" + name + "'.");
255         }
256 
257         return value;
258     }

260     protected String getParameter(Map<String, Object> parameters, String name, String defaultValue) {
261         return (String) parameters.getOrDefault(name, defaultValue);
262     }

268     protected Writer newWriter(Path path) throws IOException {
269         Charset charset = getEncoding() != null ? Charset.forName(getEncoding()) : Charset.defaultCharset();
270         return newWriter(path, charset);
271     }
272 
273     protected Writer newWriter(Path path, Charset charset) throws IOException {
274         CachingWriter cachingWriter = new CachingWriter(path, charset);
275         return new FilterWriter(cachingWriter) {
276             @Override
277             public void close() throws IOException {
278                 super.close();
279                 if (cachingWriter.isModified()) {
280                     getBuildContext().refresh(path.toFile());
281                 } else {
282                     getLogger().debug("The contents of the file " + path + " matches, skipping writing file.");
283                 }
284             }
285         };
286     }
