137     @SuppressWarnings("uncheked")
138     protected void initialize(Model model, Map<String, Object> parameters) throws ModelloException {
139         this.model = model;
140 
141         outputDirectory = new File(requireParameter(parameters, ModelloParameterConstants.OUTPUT_DIRECTORY));
142 
143         String version = requireParameter(parameters, ModelloParameterConstants.VERSION);
144 
145         generatedVersion = new Version(version);
146 
147         packageWithVersion =
148                 Boolean.parseBoolean(requireParameter(parameters, ModelloParameterConstants.PACKAGE_WITH_VERSION));
149 
150         encoding = (String) parameters.get(ModelloParameterConstants.ENCODING);
151 
152         licenseText = (List<String>) parameters.get(ModelloParameterConstants.LICENSE_TEXT);
153 
154         Optional.ofNullable(parameters.get(ModelloParameterConstants.PLURAL_EXCEPTIONS))
155                 .ifPresent(o -> PLURAL_EXCEPTIONS.putAll((Map<String, String>) o));
156     }

178     protected List<String> getHeader() {
179         List<String> header = new ArrayList<>();
180         List<String> license = getLicenseHeader();
181         if (license != null) {
182             header.addAll(license);
183         }
184         List<String> generated = getGeneratedHeader();
185         if (generated != null) {
186             header.addAll(generated);
187         }
188         return header;
189     }

191     protected List<String> getGeneratedHeader() {
192         String version = getClass().getPackage().getImplementationVersion();
193         return Arrays.asList(
194                 "=================== DO NOT EDIT THIS FILE ====================",
195                 "Generated by Modello" + ((version == null) ? "" : (' ' + version)) + ",",
196                 "any modifications will be overwritten.",
197                 "==============================================================");
198     }

204     protected boolean isClassInModel(String fieldType, Model model) {
205         try {
206             return model.getClass(fieldType, generatedVersion) != null;
207         } catch (Exception e) {
208         }
209 
210         return false;
211     }

213     /**
214      * Return the child fields of this class.
215      *
216      * @param modelClass current class
217      * @return the list of fields of this class
218      */
219     protected List<ModelField> getFieldsForClass(ModelClass modelClass) {
220         List<ModelField> fields = new ArrayList<ModelField>();
221 
222         while (modelClass != null) {
223             fields.addAll(modelClass.getFields(getGeneratedVersion()));
224 
225             String superClass = modelClass.getSuperClass();
226             if (superClass != null) {
227                 modelClass = getModel().getClass(superClass, getGeneratedVersion());
228             } else {
229                 modelClass = null;
230             }
231         }
232 
233         return fields;
234     }

248     protected String capitalise(String str) {
249         if (StringUtils.isEmpty(str)) {
250             return str;
251         }
252 
253         return new StringBuilder(str.length())
254                 .append(Character.toTitleCase(str.charAt(0)))
255                 .append(str.substring(1))
256                 .toString();
257     }

259     public static String singular(String name) {
260         if (name == null || name.isEmpty()) return name;
261 
262         String lower = name.toLowerCase();
263 
264         if (!lower.equals(name)) {
265             // we can have a case like otherArchives
266             String[] split = splitByUpperCase(name);
267             if (split != null && PLURAL_EXCEPTIONS.containsKey(split[1])) {
268                 String plural = PLURAL_EXCEPTIONS.get(split[1]);
269                 return split[0] + Character.toUpperCase(plural.charAt(0)) + plural.substring(1);
270             }
271         }
272 
273         if (PLURAL_EXCEPTIONS.containsKey(lower)) {
274             return PLURAL_EXCEPTIONS.get(lower);
275         }
276 
277         // Suffix-based rules
278         if (lower.endsWith("ies") && name.length() > 3) {
279             return name.substring(0, name.length() - 3) + "y";
280         }
281         if (lower.endsWith("aves") || lower.endsWith("lves") || lower.endsWith("rves")) {
282             return name.substring(0, name.length() - 3) + "f";
283         }
284         if (lower.endsWith("ves") && !lower.endsWith("fves")) {
285             return name.substring(0, name.length() - 3) + "fe";
286         }
287         if (lower.endsWith("zzes")) {
288             return name.substring(0, name.length() - 2);
289         }
290         if (lower.endsWith("sses")) {
291             return name.substring(0, name.length() - 2);
292         }
293         if (lower.endsWith("ses")) {
294             return name.substring(0, name.length() - 2);
295         }
296         if (lower.endsWith("ches") || lower.endsWith("shes")) {
297             return name.substring(0, name.length() - 2);
298         }
299         if (lower.endsWith("xes")) {
300             return name.substring(0, name.length() - 2);
301         }
302         if (lower.endsWith("oes")) {
303             return name.substring(0, name.length() - 1);
304         }
305         if (lower.endsWith("s") && name.length() > 1) {
306             return name.substring(0, name.length() - 1);
307         }
308 
309         return name;
310     }

312     private static String[] splitByUpperCase(String name) {
313         for (int i = name.length() - 1; i >= 0; i--) {
314             if (Character.isUpperCase(name.charAt(i))) {
315                 return new String[] {name.substring(0, i), name.substring(i).toLowerCase()};
316             }
317         }
318         return null;
319     }

321     public static String uncapitalise(String str) {
322         if (StringUtils.isEmpty(str)) {
323             return str;
324         }
325 
326         return new StringBuilder(str.length())
327                 .append(Character.toLowerCase(str.charAt(0)))
328                 .append(str.substring(1))
329                 .toString();
330     }

362     protected Writer newWriter(Path path) throws IOException {
363         Charset charset = getEncoding() != null ? Charset.forName(getEncoding()) : Charset.defaultCharset();
364         return newWriter(path, charset);
365     }

367     protected Writer newWriter(Path path, Charset charset) throws IOException {
368         CachingWriter cachingWriter = new CachingWriter(path, charset);
369         return new FilterWriter(cachingWriter) {
370             @Override
371             public void close() throws IOException {
372                 super.close();
373                 if (cachingWriter.isModified()) {
374                     getBuildContext().refresh(path.toFile());
375                 } else {
376                     getLogger().debug("The contents of the file " + path + " matches, skipping writing file.");
377                 }
378             }
379         };
380     }
