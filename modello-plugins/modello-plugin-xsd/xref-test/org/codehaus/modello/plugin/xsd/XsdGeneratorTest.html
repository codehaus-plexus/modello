39  public class XsdGeneratorTest
40      extends AbstractModelloGeneratorTest
41  {
42      public XsdGeneratorTest()
43      {
44          super( "xsd" );
45      }
46  
47      public void testXsdGenerator()
48          throws Throwable
49      {
50          ModelloCore modello = (ModelloCore) lookup( ModelloCore.ROLE );
51  
52          Model model = modello.loadModel( getXmlResourceReader( "/maven.mdo" ) );
53  
54          // check misc. properties of the model loaded
55          List<ModelClass> classesList = model.getClasses( new Version( "4.0.0" ) );
56  
57          assertEquals( 26, classesList.size() );
58  
59          ModelClass clazz = (ModelClass) classesList.get( 0 );
60  
61          assertEquals( "Model", clazz.getName() );
62  
63          ModelField extend = clazz.getField( "extend", new Version( "4.0.0" ) );
64  
65          assertTrue( extend.hasMetadata( XmlFieldMetadata.ID ) );
66  
67          XmlFieldMetadata xml = (XmlFieldMetadata) extend.getMetadata( XmlFieldMetadata.ID );
68  
69          assertNotNull( xml );
70  
71          assertTrue( xml.isAttribute() );
72  
73          assertEquals( "extender", xml.getTagName() );
74  
75          ModelField build = clazz.getField( "build", new Version( "4.0.0" ) );
76  
77          assertTrue( build.hasMetadata( XmlFieldMetadata.ID ) );
78  
79          xml = (XmlFieldMetadata) build.getMetadata( XmlFieldMetadata.ID );
80  
81          assertNotNull( xml );
82  
83          assertEquals( "builder", xml.getTagName() );
84  
85          // generate XSD file
86          Properties parameters = getModelloParameters( "4.0.0" );
87  
88          modello.generate( model, "xsd", parameters );
89  
90          //addDependency( "modello", "modello-core", "1.0-SNAPSHOT" );
91  
92          // TODO write verfier which compile generated schema : use jaxp
93  
94          //verify( "org.codehaus.modello.generator.xml.xsd.XsdVerifier", "xsd" );
95      }
