69  /**
70   * A utility class used to validate identifiers
71   * and class names
72   * @author <a href="mailto:kvisco@intalio.com">Keith Visco</a>
73   * @version $Revision$ $Date$
74   **/
75  class JNaming {
76  
77      private static final String[] keywords = {
78          "abstract",
79          "boolean",
80          "break",
81          "byte",
82          "case",
83          "catch",
84          "char",
85          "class",
86          "const",
87          "continue",
88          "default",
89          "do",
90          "double",
91          "else",
92          "extends",
93          "false",
94          "final",
95          "finally",
96          "float",
97          "for",
98          "goto",
99          "if",
100         "implements",
101         "import",
102         "instanceof",
103         "int",
104         "interface",
105         "long",
106         "native",
107         "new",
108         "null",
109         "package",
110         "private",
111         "protected",
112         "public",
113         "return",
114         "short",
115         "static",
116         "super",
117         "switch",
118         "synchronized",
119         "this",
120         "throw",
121         "throws",
122         "transient",
123         "true",
124         "try",
125         "void",
126         "volatile",
127         "while"
128     }; // -- keywords
129 
130     private JNaming() {
131         super();
132     }
133 
134     /**
135      * Returns true if the given String is a Java keyword which
136      * will cause a problem when used as a variable name
137      **/
138     public static boolean isKeyword(String name) {
139         if (name == null) return false;
140         for (String keyword : keywords) {
141             if (keyword.equals(name)) {
142                 return true;
143             }
144         }
145         return false;
146     } // -- isKeyword
147 
148     /**
149      * Returns true if the given String matches the
150      * production of a valid Java identifier
151      *
152      * @param string, the String to check the production of
153      * @return true if the given String matches the
154      * production of a valid Java name, otherwise false
155      **/
156     public static boolean isValidJavaIdentifier(String string) {
157 
158         if ((string == null) || (string.length() == 0)) return false;
159 
160         char[] chars = string.toCharArray();
161 
162         // -- make sure starting character is valid
163         if (!Character.isJavaIdentifierStart(chars[0])) return false;
164 
165         for (int i = 1; i < chars.length; i++) {
166             if (!Character.isJavaIdentifierPart(chars[i])) return false;
167         }
168         if (isKeyword(string)) return false;
169         return true;
170     } // -- isValidJavaIdentifier
171 } // -- JNaming
