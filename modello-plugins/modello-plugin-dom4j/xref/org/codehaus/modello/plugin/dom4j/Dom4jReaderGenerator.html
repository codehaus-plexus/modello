59      private boolean requiresDomSupport;
60  
61      public void generate( Model model, Properties parameters )
62          throws ModelloException
63      {
64          initialize( model, parameters );
65  
66          requiresDomSupport = false;
67  
68          try
69          {
70              generateDom4jReader();
71          }
72          catch ( IOException ex )
73          {
74              throw new ModelloException( "Exception while generating Dom4j Reader.", ex );
75          }
76      }

78      private void generateDom4jReader()
79          throws ModelloException, IOException
80      {
81          Model objectModel = getModel();
82  
83          String packageName = objectModel.getDefaultPackageName( isPackageWithVersion(), getGeneratedVersion() )
84              + ".io.dom4j";
85  
86          String unmarshallerName = getFileName( "Dom4jReader" );
87  
88          JSourceWriter sourceWriter = newJSourceWriter( packageName, unmarshallerName );
89  
90          JClass jClass = new JClass( packageName + '.' + unmarshallerName );
91          initHeader( jClass );
92          suppressAllWarnings( objectModel, jClass );
93  
94          jClass.addImport( "java.io.InputStream" );
95          jClass.addImport( "java.io.IOException" );
96          jClass.addImport( "java.io.Reader" );
97          jClass.addImport( "java.net.URL" );
98          jClass.addImport( "java.util.Date" );
99          jClass.addImport( "java.util.Locale" );
100         jClass.addImport( "java.text.DateFormat" );
101         jClass.addImport( "java.text.ParsePosition" );
102         jClass.addImport( "java.util.Iterator" );
103         jClass.addImport( "org.dom4j.Attribute" );
104         jClass.addImport( "org.dom4j.Document" );

107         jClass.addImport( "org.dom4j.Node" );
108         jClass.addImport( "org.dom4j.io.SAXReader" );
109 
110         addModelImports( jClass, null );
111 
112         ModelClass root = objectModel.getClass( objectModel.getRoot( getGeneratedVersion() ), getGeneratedVersion() );
113         JClass rootType = new JClass( root.getName() );
114

178

186 
187         unmarshall.addParameter( new JParameter( new JClass( "InputStream" ), "stream" ) );

192

194

198

200

202

204 
205         unmarshall = new JMethod( "read", rootType, null );
206 
207         unmarshall.addParameter( new JParameter( new JClass( "InputStream" ), "stream" ) );
208 
209         unmarshall.addException( new JClass( "IOException" ) );
210         unmarshall.addException( new JClass( "DocumentException" ) );

214         sc.add( "return read( stream, true );" );
215 
216         jClass.addMethod( unmarshall );

219         // Write the read(URL[,boolean]) methods which will do the unmarshalling.
220         // ----------------------------------------------------------------------

222         unmarshall = new JMethod( "read", rootType, null );

224         unmarshall.addParameter( new JParameter( new JClass( "URL" ), "url" ) );
225         unmarshall.addParameter( new JParameter( JType.BOOLEAN, "strict" ) );

227         unmarshall.addException( new JClass( "IOException" ) );
228         unmarshall.addException( new JClass( "DocumentException" ) );
229 
230         sc = unmarshall.getSourceCode();
231 
232         sc.add( "SAXReader parser = new SAXReader();" );
233 
234         sc.add( "Document document = parser.read( url );" );

236         sc.add( "return read( document, strict );" );
237 
238         jClass.addMethod( unmarshall );
239 
240         // ----------------------------------------------------------------------
241 
242         unmarshall = new JMethod( "read", rootType, null );
243 
244         unmarshall.addParameter( new JParameter( new JClass( "URL" ), "url" ) );
245 
246         unmarshall.addException( new JClass( "IOException" ) );
247         unmarshall.addException( new JClass( "DocumentException" ) );
248 
249         sc = unmarshall.getSourceCode();
250 
251         sc.add( "return read( url, true );" );
252 
253         jClass.addMethod( unmarshall );
254 
255         // ----------------------------------------------------------------------
256         // Write the class parsers
257         // ----------------------------------------------------------------------
258 
259         writeAllClassesParser( objectModel, jClass );

261         // ----------------------------------------------------------------------

263         // ----------------------------------------------------------------------

265         writeHelpers( jClass );

267         if ( requiresDomSupport )

271         }
272 
273         // ----------------------------------------------------------------------
274         //
275         // ----------------------------------------------------------------------
276 
277         jClass.print( sourceWriter );
278 
279         sourceWriter.close();
280     }

282     private void writeAllClassesParser( Model objectModel, JClass jClass )
283     {
284         ModelClass root = objectModel.getClass( objectModel.getRoot( getGeneratedVersion() ), getGeneratedVersion() );
285 
286         for ( ModelClass clazz : getClasses( objectModel ) )
287         {
288             writeClassParser( clazz, jClass, root.getName().equals( clazz.getName() ) );
289         }
290     }

292     private void writeClassParser( ModelClass modelClass, JClass jClass, boolean rootElement )
293     {
294         String className = modelClass.getName();
295 
296         String capClassName = capitalise( className );
297 
298         String uncapClassName = uncapitalise( className );
299 
300         JMethod unmarshall = new JMethod( "parse" + capClassName, new JClass( className ), null );
301         unmarshall.getModifiers().makePrivate();
302 
303         unmarshall.addParameter( new JParameter( new JClass( "String" ), "tagName" ) );
304         unmarshall.addParameter( new JParameter( new JClass( "Element" ), "element" ) );
305         unmarshall.addParameter( new JParameter( JType.BOOLEAN, "strict" ) );
306 
307         unmarshall.addException( new JClass( "IOException" ) );
308         unmarshall.addException( new JClass( "DocumentException" ) );
309 
310         JSourceCode sc = unmarshall.getSourceCode();
311 
312         sc.add( className + " " + uncapClassName + " = new " + className + "();" );
313 
314         ModelField contentField = null;
315 
316         List<ModelField> modelFields = getFieldsForXml( modelClass, getGeneratedVersion() );
317 
318         // read all XML attributes first
319         for ( ModelField field : modelFields )
320         {
321             XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
322 
323             if ( xmlFieldMetadata.isAttribute() )
324             {
325                 String tagName = xmlFieldMetadata.getTagName();
326                 if ( tagName == null )
327                 {
328                     tagName = field.getName();
329                 }
330 
331                 sc.add( "if ( element.attributeValue( \"" + tagName + "\" ) != null  )" );
332                 sc.add(  "{" );
333                 sc.indent();
334 
335                 writePrimitiveField( field, field.getType(), uncapClassName, "set" + capitalise( field.getName() ), sc,
336                                      jClass, "element", "childElement" );
337 
338                 sc.unindent();
339                 sc.add( "}" );
340             }
341             // TODO check if we have already one with this type and throws Exception
342             if ( xmlFieldMetadata.isContent() )
343             {
344                 contentField = field;
345             }
346         }
347 
348         if ( rootElement )
349         {
350             sc.add( "if ( strict )" );
351             sc.add( "{" );
352             sc.indent();
353 
354             sc.add( "if ( !element.getName().equals( tagName ) )" );
355             sc.add( "{" );
356             sc.addIndented(
357                 "throw new DocumentException( \"Error parsing model: root element tag is '\" + element.getName() + \"' instead of '\" + tagName + \"'\" );" );
358             sc.add( "}" );
359 
360             sc.unindent();
361             sc.add( "}" );
362         }
363 
364         if ( contentField != null )
365         {
366             writePrimitiveField( contentField, contentField.getType(), uncapClassName,
367                                  "set" + capitalise( contentField.getName() ), sc, jClass, null, "element" );
368         }
369         else
370         {
371             sc.add( "java.util.Set parsed = new java.util.HashSet();" );
372 
373             sc.add( "for ( Iterator i = element.nodeIterator(); i.hasNext(); )" );
374             sc.add( "{" );
375             sc.indent();
376 
377             sc.add( "Node node = (Node) i.next();" );
378 
379             sc.add( "if ( node.getNodeType() == Node.ELEMENT_NODE )" );
380             // TODO: attach other NodeTypes to model in some way
381             sc.add( "{" );
382             sc.indent();
383 
384             sc.add( "Element childElement = (Element) node;" );
385 
386             boolean addElse = false;
387 
388             for ( ModelField field : modelFields )
389             {
390                 XmlFieldMetadata xmlFieldMetadata = (XmlFieldMetadata) field.getMetadata( XmlFieldMetadata.ID );
391 
392                 if ( !xmlFieldMetadata.isAttribute() )
393                 {
394                     processField( field, xmlFieldMetadata, addElse, sc, uncapClassName, jClass );
395 
396                     addElse = true;
397                 }
398             }
399 
400             if ( addElse )
401             {
402                 sc.add( "else" );
403 
404                 sc.add( "{" );
405                 sc.indent();
406             }
407 
408             sc.add( "checkUnknownElement( childElement, strict );" );
409 
410             if ( addElse )
411             {
412                 sc.unindent();
413                 sc.add( "}" );
414             }
415 
416             sc.unindent();
417             sc.add( "}" );
418 
419             sc.unindent();
420             sc.add( "}" );
421         }
422 
423         sc.add( "return " + uncapClassName + ";" );
424 
425         jClass.addMethod( unmarshall );
426     }

428     /**
429      * Generate code to process a field represented as an XML element.
430      *
431      * @param field the field to process
432      * @param xmlFieldMetadata its XML metadata
433      * @param addElse add an <code>else</code> statement before generating a new <code>if</code>
434      * @param sc the method source code to add to
435      * @param objectName the object name in the source
436      * @param jClass the generated class source file
437      */
438     private void processField( ModelField field, XmlFieldMetadata xmlFieldMetadata, boolean addElse, JSourceCode sc,
439                                String objectName, JClass jClass )
440     {
441         String fieldTagName = resolveTagName( field, xmlFieldMetadata );
442 
443         String capFieldName = capitalise( field.getName() );
444 
445         String singularName = singular( field.getName() );
446 
447         String alias;
448         if ( StringUtils.isEmpty( field.getAlias() ) )
449         {
450             alias = "null";
451         }
452         else
453         {
454             alias = "\"" + field.getAlias() + "\"";
455         }
456 
457         String tagComparison = ( addElse ? "else " : "" )
458             + "if ( checkFieldWithDuplicate( childElement, \"" + fieldTagName + "\", " + alias + ", parsed ) )";
459 
460         if ( field instanceof ModelAssociation )
461         {
462             ModelAssociation association = (ModelAssociation) field;
463 
464             String associationName = association.getName();
465 
466             if ( association.isOneMultiplicity() )
467             {
468                 sc.add( tagComparison );
469 
470                 sc.add( "{" );
471                 sc.addIndented( objectName + ".set" + capFieldName + "( parse" + association.getTo() + "( \""
472                                 + fieldTagName + "\", childElement, strict ) );" );
473                 sc.add( "}" );
474             }
475             else
476             {
477                 //MANY_MULTIPLICITY
478 
479                 XmlAssociationMetadata xmlAssociationMetadata =
480                     (XmlAssociationMetadata) association.getAssociationMetadata( XmlAssociationMetadata.ID );
481 
482                 String valuesTagName = resolveTagName( fieldTagName, xmlAssociationMetadata );
483 
484                 String type = association.getType();
485 
486                 if ( ModelDefault.LIST.equals( type ) || ModelDefault.SET.equals( type ) )
487                 {
488                     boolean wrappedItems = xmlAssociationMetadata.isWrappedItems();
489 
490                     if ( wrappedItems )
491                     {
492                         sc.add( tagComparison );
493 
494                         sc.add( "{" );
495                         sc.indent();
496 
497                         sc.add( type + " " + associationName + " = " + association.getDefaultValue() + ";" );
498 
499                         sc.add( objectName + ".set" + capFieldName + "( " + associationName + " );" );
500 
501                         sc.add( "for ( Iterator j = childElement.nodeIterator(); j.hasNext(); )" );
502 
503                         sc.add( "{" );
504                         sc.indent();
505 
506                         sc.add( "Node n = (Node) j.next();" );
507 
508                         sc.add( "if ( n.getNodeType() == Node.ELEMENT_NODE )" );
509                         // TODO: track the whitespace in the model (other NodeTypes)
510 
511                         sc.add( "{" );
512                         sc.indent();
513 
514                         sc.add( "Element listElement = (Element) n;" );
515 
516                         sc.add( "if ( \"" + valuesTagName + "\".equals( listElement.getName() ) )" );
517 
518                         sc.add( "{" );
519                         sc.indent();
520                     }
521                     else
522                     {
523                         sc.add( ( addElse ? "else " : "" )
524                             + "if ( \"" + valuesTagName + "\".equals( childElement.getName() ) )" );
525 
526                         sc.add( "{" );
527                         sc.indent();
528 
529                         sc.add( "Element listElement = childElement;" );
530 
531                         sc.add( type + " " + associationName + " = " + objectName + ".get" + capFieldName + "();" );
532 
533                         sc.add( "if ( " + associationName + " == null )" );
534 
535                         sc.add( "{" );
536                         sc.indent();
537 
538                         sc.add( associationName + " = " + association.getDefaultValue() + ";" );
539 
540                         sc.add( objectName + ".set" + capFieldName + "( " + associationName + " );" );
541 
542                         sc.unindent();
543                         sc.add( "}" );
544                     }
545 
546                     if ( isClassInModel( association.getTo(), field.getModelClass().getModel() ) )
547                     {
548                         sc.add( associationName + ".add( parse" + association.getTo() + "( \"" + valuesTagName
549                             + "\", listElement, strict ) );" );
550                     }
551                     else
552                     {
553                         writePrimitiveField( association, association.getTo(), associationName, "add", sc, jClass,
554                                              "childElement", "listElement" );
555                     }
556 
557                     if ( wrappedItems )
558                     {
559                         sc.unindent();
560                         sc.add( "}" );
561 
562                         sc.add( "else" );
563 
564                         sc.add( "{" );
565                         sc.add( "}" );
566 
567                         sc.unindent();
568                         sc.add( "}" );
569 
570                         sc.unindent();
571                         sc.add( "}" );
572 
573                         sc.unindent();
574                         sc.add( "}" );
575                     }
576                     else
577                     {
578                         sc.unindent();
579                         sc.add( "}" );
580                     }
581                 }
582                 else
583                 {
584                     //Map or Properties
585 
586                     sc.add( tagComparison );
587 
588                     sc.add( "{" );
589                     sc.indent();
590 
591                     if ( xmlAssociationMetadata.isMapExplode() )
592                     {
593                         sc.add( "for ( Iterator j = childElement.nodeIterator(); j.hasNext(); )" );
594 
595                         sc.add( "{" );
596                         sc.indent();
597 
598                         sc.add( "Node n = (Node) j.next();" );
599 
600                         sc.add( "if ( n.getNodeType() == Node.ELEMENT_NODE )" );
601                         // TODO: track the whitespace in the model (other NodeTypes)
602 
603                         sc.add( "{" );
604                         sc.indent();
605 
606                         sc.add( "Element listElement = (Element) n;" );
607 
608                         sc.add( "if ( \"" + valuesTagName + "\".equals( listElement.getName() ) )" );
609 
610                         sc.add( "{" );
611                         sc.indent();
612 
613                         sc.add( "String key = null;" );
614 
615                         sc.add( "String value = null;" );
616 
617                         sc.add( "//" + xmlAssociationMetadata.getMapStyle() + " mode." );
618 
619                         sc.add( "for ( Iterator k = listElement.nodeIterator(); k.hasNext(); )" );
620 
621                         sc.add( "{" );
622                         sc.indent();
623 
624                         sc.add( "Node nd = (Node) k.next();" );
625 
626                         sc.add( "if ( nd.getNodeType() == Node.ELEMENT_NODE )" );
627                         // TODO: track the whitespace in the model (other NodeTypes)
628 
629                         sc.add( "{" );
630                         sc.indent();
631 
632                         sc.add( "Element propertyElement = (Element) nd;" );
633 
634                         sc.add( "if ( \"key\".equals( propertyElement.getName() ) )" );
635 
636                         sc.add( "{" );
637                         sc.addIndented( "key = propertyElement.getText();" );
638                         sc.add( "}" );
639 
640                         sc.add( "else if ( \"value\".equals( propertyElement.getName() ) )" );
641 
642                         sc.add( "{" );
643                         sc.addIndented( "value = propertyElement.getText()"
644                                         + ( xmlFieldMetadata.isTrim() ? ".trim()" : "" ) + ";" );
645                         sc.add( "}" );
646 
647                         sc.add( "else" );
648 
649                         sc.add( "{" );
650                         sc.add( "}" );
651 
652                         sc.unindent();
653                         sc.add( "}" );
654 
655                         sc.unindent();
656                         sc.add( "}" );
657 
658                         sc.add( objectName + ".add" + capitalise( singularName ) + "( key, value );" );
659 
660                         sc.unindent();
661                         sc.add( "}" );
662 
663                         sc.unindent();
664                         sc.add( "}" );
665 
666                         sc.unindent();
667                         sc.add( "}" );
668                     }
669                     else
670                     {
671                         //INLINE Mode
672 
673                         sc.add( "for ( Iterator j = childElement.nodeIterator(); j.hasNext(); )" );
674 
675                         sc.add( "{" );
676                         sc.indent();
677 
678                         sc.add( "Node n = (Node) j.next();" );
679 
680                         sc.add( "if ( n.getNodeType() == Node.ELEMENT_NODE )" );
681                         // TODO: track the whitespace in the model (other NodeTypes)
682 
683                         sc.add( "{" );
684                         sc.indent();
685 
686                         sc.add( "Element listElement = (Element) n;" );
687 
688                         sc.add( "String key = listElement.getName();" );
689 
690                         sc.add( "String value = listElement.getText()"
691                                 + ( xmlFieldMetadata.isTrim() ? ".trim()" : "" ) + ";" );
692 
693                         sc.add( objectName + ".add" + capitalise( singularName ) + "( key, value );" );
694 
695                         sc.unindent();
696                         sc.add( "}" );
697 
698                         sc.unindent();
699                         sc.add( "}" );
700                     }
701 
702                     sc.unindent();
703                     sc.add( "}" );
704                 }
705             }
706         }
707         else
708         {
709             sc.add( tagComparison );
710 
711             sc.add( "{" );
712             sc.indent();
713 
714             //ModelField
715             writePrimitiveField( field, field.getType(), objectName, "set" + capitalise( field.getName() ), sc,
716                                  jClass, "element", "childElement" );
717 
718             sc.unindent();
719             sc.add( "}" );
720         }
721     }

723     private void writePrimitiveField( ModelField field, String type, String objectName, String setterName,
724                                       JSourceCode sc, JClass jClass, String parentElementName, String childElementName )
725     {
